openapi: 3.0.1
info:

      summary: 官方格式参数生成

            schema:

                              - url
                            description: url或者base64
                            title: 输入给模型的图片对象
                          role:
                            type: string
                            title: 输入给模型的图片对象
                            description: |-
                              first_frame 首帧图
                              last_frame 尾帧图
                              reference_image 参考图
                              reference_video 参考视频
                              reference_audio 参考音频
                          video_url:
                            type: object
                            properties:
                              url:
                                type: string
                            required:
                              - url
                            x-apifox-orders:
                              - url
                            description: 视频的公网 URL
                          audio_url:
                            type: object
                            properties:
                              url:
                                type: string
                            required:
                              - url
                            x-apifox-orders:
                              - url
                            description: 音频 URL 、音频 Base64 编码
                        required:
                          - type
                          - image_url
                          - role
                        x-apifox-orders:
                          - type
                          - image_url
                          - role
                          - video_url
                          - audio_url
                      description: 输入给模型，生成视频的信息，支持文本、图片、音频、视频
                    generate_audio:
                      type: boolean
                      description: 生成有声视频
                      title: 生成有声视频
                    ratio:
                      type: string
                      title: 生成视频的宽高比例
                    duration:
                      type: integer
                      title: 时长
                    watermark:
                      type: boolean
                      title: 输出时长
                  required:
                    - content
                    - generate_audio
                    - ratio
                    - duration
                    - watermark
                  x-apifox-orders:
                    - content
                    - generate_audio
                    - ratio
                    - duration
                    - watermark
                  title: 媒体信息包装data
                  description: 整个数据结构同步给官网接口
              required:
                - model
                - prompt
                - metadata
              x-apifox-orders:
                - model
                - prompt
                - metadata
            example:

                秒：快速切镜，你的手将苹果块投入雪克杯，加入冰块与茶底，用力摇晃，冰块碰撞声与摇晃声卡点轻快鼓点，背景音：「鲜切现摇」；4-6
                秒：第一人称成品特写，分层果茶倒入透明杯，你的手轻挤奶盖在顶部铺展，在杯身贴上粉红包标，镜头拉近看奶盖与果茶的分层纹理；6-8
                秒：第一人称手持举杯，你将图片2中的果茶举到镜头前（模拟递到观众面前的视角），杯身标签清晰可见，背景音「来一口鲜爽」，尾帧定格为图片2。背景声音统一为女生音色。
              metadata:
                content:
                  - type: image_url
                    image_url:
                      url: >-
                        https://ark-project.tos-cn-beijing.volces.com/doc_image/r2v_tea_pic1.jpg
                    role: reference_image
                  - type: image_url
                    image_url:
                      url: >-
                        https://ark-project.tos-cn-beijing.volces.com/doc_image/r2v_tea_pic2.jpg
                    role: reference_image
                  - type: video_url
                    video_url:
                      url: >-
                        https://ark-project.tos-cn-beijing.volces.com/doc_video/r2v_tea_video1.mp4
                    role: reference_video
                  - type: audio_url
                    audio_url:
                      url: >-
                        https://ark-project.tos-cn-beijing.volces.com/doc_audio/r2v_tea_audio1.mp3
                    role: reference_audio
                generate_audio: true
                ratio: '16:9'
                duration: 11
                watermark: true
      responses:
        '200':
          description: ''
          content:
            application/json:
              schema:
                type: object
                properties: {}
          headers: {}
          x-apifox-name: 成功
      security: []

      x-apifox-status: released
